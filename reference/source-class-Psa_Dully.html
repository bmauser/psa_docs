 32 /**
 33  * Simple template engine class.
 34  *
 35  * This class has few methods that are similar to those in {@link http://www.smarty.net Smarty}
 36  * template engine.
 37  *
 38  * All that Dully does is putting values from an associative array into the local namespace and
 39  * includes .php (template) file. There are no special template tags, just use PHP code
 40  * blocks to echo values. Templates are ordinary PHP files and you can write them
 41  * as any other .php script, but keep in mind that point of the template engine
 42  * is separating business logic from presentation.
 43  * So, you should not put any logic (calculations, getting data from a database ...)
 44  * into the templates. With {@link http://www.smarty.net Smarty} that is easier to achieve,
 45  * and it has many advanced features so I suggest that you use Smarty as a template engine.
 46  * If you need a simple and fast template engine or don't want to learn Smarty's tags, you can
 47  * use Dully.
 48  * Dully class doesn't include anything from PSA so you can use this class as template engine
 49  * in any PHP application.
 50  *
 51  * I named this class Dully as opposite to Smarty. This class is inspired from 
 52  * {@link http://www.massassi.com/php/articles/template_engines/ here}
 53  * and there are some interesting thoughts about template engines.
 54  *
 55  * <br><b>Usage examples:</b>
 56  *
 57  * <br><b>1)</b> Simple example with one template file:
 58  *
 59  * Template file <samp>template.tpl</samp>:
 60  * <code>
 61  * This is a text from template file template.php<br/>
 62  * My car is <?php echo $my_car_color ?>.<br/>
 63  * Something else: <?php echo $some_name ?>
 64  * </code>
 65  *
 66  * .php file:
 67  * <code>
 68  * <?php
 69  *
 70  * // make a new Psa_Dully object
 71  * $Psa_Dully = new Psa_Dully('path/to/dir/with/templates');
 72  *
 73  * // assign some data for template
 74  * $Psa_Dully->assign('my_car_color', 'Black');
 75  * $Psa_Dully->assign('some_name', 'bla bla bla');
 76  *
 77  * // display fetched template
 78  * $Psa_Dully->display('template.tpl');
 79  *
 80  * ?>
 81  * </code>
 82  *
 83  * The printout of the above .php script will be:
 84  * <pre>
 85  * This is a text from template file template.php
 86  * My car is Black.
 87  * Something else: bla bla bla
 88  * </pre>
 89  *
 90  * <b>Note:</b> you can write <kbd><?= $some_name ?></kbd> instead of <kbd><?php echo $some_name ?></kbd>.
 91  * Use of this shortcut requires {@link http://www.php.net/manual/en/ini.core.php#ini.short-open-tag short_open_tag}
 92  * PHP ini option to be on.
 93  *
 94  * <br><b>2)</b> You can put one rendered template into another:
 95  *
 96  * template file <samp>template1.tpl</samp>:
 97  * <code>
 98  * This is a text from template1.php<br/>
 99  * <?php echo $var1 ?><br/>
100  * <?php echo $var2 ?>
101  * </code>
102  *
103  * template file <samp>template2.tpl</samp>:
104  * <code>
105  * This is a text from template2.php
106  * </code>
107  *
108  * .php file:
109  * <code>
110  * <?php
111  *
112  * // make a new Psa_Dully object
113  * $Psa_Dully = new Psa_Dully('path/to/dir/with/templates');
114  *
115  * // put the content of fetched template template2.tpl in the variable $var1
116  * $var1 = $Psa_Dully->fetch('template2.tpl');
117  *
118  * // assign values for var1 and var2
119  * $Psa_Dully->assign('var1', $var1);
120  * $Psa_Dully->assign('var2', 'bla bla bla');
121  *
122  * // display template template1.tpl
123  * $Psa_Dully->display('template1.tpl');
124  *
125  * ?>
126  * </code>
127  *
128  * The above example will output:
129  * <pre>
130  * This is a text from template1.php
131  * This is a text from template2.php
132  * bla bla bla
133  * </pre>
134  *
135  * <br><b>3)</b> In this kind of templates it's nice to use alternative PHP syntax for control structures.
136  * See details in {@link http://www.php.net/manual/en/control-structures.alternative-syntax.php PHP documentation}.
137  * <code>
138  * <? if ($variable == 'abcd'): ?>
139  *  <h1>this is something</h1>
140  *  <p>
141  *      bla bla
142  *  </p>
143  * <? endif ?>
144  *
145  * <? foreach ($array as $value): ?>
146  *  <div><?= $value['email']) ?></div>
147  *  <div><?= $value['phone_number']) ?></div>
148  * <? endforeach ?>
149  * </code>
150  *
151  *
152  * @see Psa_Dully_View
153  * @see Psa_Smarty
154  */
155 class Psa_Dully{
156 
157     /**
158      * Array that holds template values.
159      *
160      * @var array
161      * @ignore
162      */
163     protected $template_values = array();
164 
165 
166     /**
167      * Path to folder with templates.
168      * Without / at the end. It is set by constructor.
169      *
170      * @var string
171      * @ignore
172      */
173     protected $template_dir;
174 
175 
176     /**
177      * Constructor.
178      *
179      * @param string $template_dir Path to the the directory with templates.
180      */
181     function __construct($template_dir){
182         $this->template_dir = $template_dir;
183     }
184 
185 
186     /**
187      * Assigns values to the templates.
188      * See examples in {@link Psa_Dully} class description.
189      *
190      * @param string $name The name of the variable being assigned.
191      * @param mixed $value The value being assigned.
192      * @see display()
193      * @see fetch()
194      */
195     function assign($name, $value){
196         $this->template_values[$name] = $value;
197     }
198 
199 
200     /**
201      * Returns the fetched template as a string.
202      * See examples in {@link Psa_Dully} class description.
203      *
204      * @param string $template_file Template file name or relative path from the path passed to the constructor.
205      * @return string Fetched (rendered) template.
206      * @see display()
207      */
208     function fetch($template_file){
209 
210         // just to be sure that variable will not be overwritten with extract()
211         $sdf33saf2342as8dmm32 = $template_file;
212 
213         // extract the template_values to local namespace
214         extract($this->template_values);
215 
216         // start output buffering
217         ob_start();
218 
219         // include the file
220         include $this->template_dir . '/' . $sdf33saf2342as8dmm32;
221 
222         // get the contents and clean the buffer
223         return ob_get_clean();
224     }
225 
226 
227     /**
228      * Outputs the fetched template.
229      *
230      * Just prints output from {@link fetch()} method.
231      *
232      * @param string $template_file Template file name or relative path from path passed to the constructor.
233      * @see fetch()
234      */
235     function display($template_file){
236         echo $this->fetch($template_file);
237     }
238 }
