 32 /**
 33  * This class extends PHP Exception and adds method for logging.
 34  * 
 35  * Other PSA exception classes extend this class. You can also extend this class 
 36  * for your custom exceptions. If logs are enabled in {@link config.php} file 
 37  * exception messages will be logged.
 38  */
 39 class Psa_Exception extends Exception{
 40 
 41 
 42     /**
 43      * Constructor.
 44      *
 45      * Calls {@link log()} method which writes message to log.
 46      *
 47      * @param string|array $message exception message or associative
 48      * array with values for {@link log()} method.
 49      * @param string $code user defined exception code
 50      * @param bool $write_log set to false if you don't want to write exception to log
 51      */
 52     public function __construct($message = '', $code = 0, $write_log = true){
 53 
 54         if(is_array($message))
 55             $exception_message = $message['message'];
 56         else
 57             $exception_message = $message;
 58 
 59         // make sure everything is assigned properly
 60         parent::__construct($exception_message, (int)$code); //$code can be string here
 61 
 62         // write log
 63         if($write_log)
 64             $this->log($message);
 65     }
 66 
 67 
 68     /**
 69      * Writes a log message.
 70      *
 71      * @param string|array $log_message message to be logged or associative
 72      * array with values for {@link Psa_Logger::log()} method. If you pass
 73      * array some elements, if not set, will get default values.
 74      * Default values are:
 75      * - for <kbd>'function'</kbd> element is return from <kbd>getTraceAsString()</kbd>
 76      *   exception method
 77      * - for <kbd>'level'</kbd> element is 1
 78      * - for <kbd>'type'</kbd> element is <kbd>'exception'</kbd>.
 79      *
 80      * @see Psa_Logger::log()
 81      */
 82     protected function log($log_message){
 83 
 84         // if logging is enabled
 85         if(Psa_Registry::get_instance()->PSA_CFG['logging']['max_log_level'] >= 1){
 86             // parameters for Psa_Logger::log() method
 87             if(is_array($log_message)){
 88                 $log_data = &$log_message;
 89             }
 90             else
 91                 $log_data['message'] = $log_message;
 92 
 93             if(!@$log_data['function'])
 94                 $log_data['function'] = $this->getTraceAsString();
 95 
 96             if(!@$log_data['level'])
 97                 $log_data['level'] = 1;
 98 
 99             if(!@$log_data['type'])
100                 $log_data['type'] = get_class($this);
101 
102             Psa_Logger::get_instance()->log($log_data);
103         }
104     }
105 }
