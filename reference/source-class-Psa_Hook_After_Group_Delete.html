32 /**
33  * Abstract class for <i>Psa_Hook_After_Group_Delete</i> hooks.
34  *
35  * You can write <i>Psa_Hook_After_Group_Delete</i> hooks by extending this class.
36  * After the group is successfully deleted from the database with {@link psa_delete_group()} function,
37  * <kbd>psa_main()</kbd> method of the hook object that extends this class will be called.
38  * <br><br><b>Example:</b>
39  *
40  * <code>
41  * <?php
42  * class my_hook extends Psa_Hook_After_Group_Delete{
43  *
44  *     // you have to define psa_main() method in your child class
45  *     function psa_main($group){
46  *
47  *         // delete folder associated with the group
48  *         rmdir ('/var/groups/group_' . $group);
49  *     }
50  * }
51  * ?>
52  * </code>
53  *
54  * When this example hook is registered, after group deletion with {@link psa_delete_group()} function,
55  * folder associated with the group will be deleted.
56  *
57  * <b>Note:</b> the row in the database will be deleted before calling hook's <kbd>psa_main()</kbd> method.
58  * So, you cannot read other information about the group from the database because database record doesn't
59  * exist any more and all information you have is group ID or name passed as argument to <kbd>psa_main()</kbd> method.
60  *
61  * <b>Note:</b> the argument passed to the <kbd>psa_main()</kbd> method can be integer (group ID) or string (group name).
62  * This depends on what you pass to {@link psa_delete_group()} function.
63  *
64  *
65  * @see Psa_Group
66  * @see Psa_Hook_Before_Group_Delete
67  */
68 abstract class Psa_Hook_After_Group_Delete extends Psa_Model{
69 
70 
71     /**
72      * This method will be called after the group is successfully deleted from the database with {@link psa_delete_group()} function.
73      *
74      * @param int|string $group ID or name of the deleted group as passed to {@link psa_delete_group()} function.
75      * @see psa_delete_group()
76      */
77     abstract protected function psa_main($group);
78 }
