 38  * This class implements {@link http://en.wikipedia.org/wiki/Singleton_pattern singleton pattern}
 39  * to ensure that there is only one instance of the {@link Psa_Logger} object and
 40  * to allow that this single instance is easily accessible from any scope.
 41  * You should not make instance of {@link Psa_Logger} object with the <kbd>new</kbd> operator
 42  * instead call static method {@link get_instance()}.
 43  *
 44  * <br><b>Examples:</b>
 45  *
 46  * <br><b>1)</b> Log a single message.
 47  * <code>
 48  * Psa_Logger::get_instance()->log("Log message");

 50  *
 51  * <br><b>2)</b> If you want to extend this class you can write your child class like this:

 53  * <?php

120     /**
121      * Writes logs.
122      *
123      * Writes a log record to database or file. This depends on settings (<var>$PSA_CFG['logging']</var>)
124      * in {@link config.php}. First argument can be array or string. If is a string it is written to the log.
125      * If it is array some other values in log record can be set. Here is example array with all
126      * available elements and example values that can be used as first argument for this method.
127      * <pre>
128      * Array
129      * (
130      *     [user_id] => 5                // id of user
131      *     [username] => john            // username
132      *     [group_id] => 3               // id of group
133      *     [groupname] => test           // group name
134      *     [message] => new user created // log message
135      *     [function] => Psa_User::save  // method or function name that created log
136      *     [level] => 2                  // Log level, default 1. Integer. If greater than <var>$PSA_CFG['logging']['level']</var> log will not be saved.
137      *     [type] => general             // Log message type. For exceptions this is exception class name. Custom string value.
138      * )
139      * </pre>
140      * All array elements are optional.
141      *
142      * <b>Example:</b>
143      *
144      * <code>
145      * <?php
146      *
147      * // set data to log
148      * $log_data['user_id']  = 123;
149      * $log_data['username'] = 'abc';
150      * $log_data['message']  = 'test message';
151      * $log_data['function'] = 'some_method';
152      * $log_data['level']    = 1;
153      * $log_data['type']     = 'some_type';
154      *
155      * // write log
156      * Psa_Logger::get_instance()->log($log_data);
157      *
158      * ?>
159      * </code>
160      *
161      * @param array|string $log_data array with data to be logged or log message only. If is a string, that string
162      * is used as message to be logged.
163      * @param int|string $log_storage storage for the log. This is index of <var>$PSA_CFG['logging']['storage']</var>
164      * array in {@link config.php}. Default is 'psa_default'.
165      * @return int 0 for failure, 1 for success, -1 logging is disabled or the log level for message is set too low
166      * (bigger number than <var>$PSA_CFG['logging']['max_log_level'] </var>).
167      * @see get_instance()
168      * @see format_file_log()
169      * @see format_database_log()
170      * @see config.php
171      */
172     public function log($log_data, $log_storage = 'psa_default'){
173 
174         // config array
175         $PSA_CFG = Psa_Registry::get_instance()->PSA_CFG;
176 
177         // if $log_data is not array
178         if(!is_array($log_data)){
179             $msg = (string)$log_data;
180             $log_data = array();
181             $log_data['message'] = $msg;
182         }
183 
184         // default log level
185         if(!isset($log_data['level']) or !$log_data['level'])
186             $log_data['level'] = 1;
187 
188         // default log type
189         if(!isset($log_data['type']) or !$log_data['type'])
190             $log_data['type'] = null;
191 
192         // if not set user data, use data from $_SESSION['psa_current_user_data']
193         if(!@$log_data['user_id'] && !@$log_data['username'] && isset($_SESSION['psa_current_user_data']) && $_SESSION['psa_current_user_data'] instanceof Psa_User){
194             $log_data['user_id'] = $_SESSION['psa_current_user_data']['id'];
195             $log_data['username'] = $_SESSION['psa_current_user_data']['username'];
196         }
197 
198 
199         // if logging is enabled
200         if($PSA_CFG['logging']['max_log_level'] >= $log_data['level']){
201 
202             // write log to file
203             if($PSA_CFG['logging']['storage'][$log_storage]['type'] == 'file'){
204 
205                 // format log message
206                 $message = $this->format_file_log($log_data);
207 
208                 // open log file
209                 if (!$handle = fopen($PSA_CFG['logging']['storage'][$log_storage]['target'], 'a')){
210                     trigger_error("Cannot open log file: {$PSA_CFG['logging']['storage'][$log_storage]['target']}");
211                     return 0;
212                 }
213 
214                 // write to file
215                 else{
216                     // if error writing to file
217                     if (!fwrite($handle, $message)){
218                         trigger_error("Cannot write to log file: {$PSA_CFG['logging']['storage'][$log_storage]['target']}");
219                         fclose($handle);
220                         return 0;
221                     }
222 
223                     // close file
224                     fclose($handle);
225                     return 1;
226                 }
227             }
228             // write log to database
229             else if($PSA_CFG['logging']['storage'][$log_storage]['type'] == 'database'){
230 
231                 if(!($this->database instanceof Psa_PDO)){
232 
233                     // should new database connection be opened
234                     if(@$PSA_CFG['logging']['new_database_connection']){
235                         $psa_registry = Psa_Registry::get_instance();
236                         if(!(@$psa_registry->psa_log_database_connection instanceof Psa_PDO)){
237                             $psa_registry->psa_log_database_connection = new Psa_PDO();
238                         }
239                         $this->database = $psa_registry->psa_log_database_connection;
240                     }
241                     else{
242                         // existing database connection
243                         $this->database = Psa_Registry::get_instance()->psa_database;
244                     }
245                 }
246 
247                 // prepare database query if not already prepared
248                 if(!$this->prepared_query){
249                     $this->prepared_query = $this->database->prepare($this->format_database_log_query($log_storage));
250                 }
251 
252                 // run query against the database
253                 try{
254                     $this->database->execute($this->format_database_log_query_params($log_data), $this->prepared_query, 1);
255                 }
256                 catch(PDOException $e){
257 
258                     $message = "Unable to write log message to database table '{$PSA_CFG['logging']['storage'][$log_storage]['target']}' " . $e->getMessage();
259 
260                     // log message cannot be written into the database if there is problem with database connection
261                     if($PSA_CFG['logging']['storage']['psa_default']['type'] == 'database')
262                         throw new Psa_Fatal_Error_Exception($message, $message, $e->getCode(), false);
263                     else
264                         throw new Psa_Fatal_Error_Exception($message, $message, $e->getCode());
265                 }
266 
267                 return 1;
268             }
269         }
270         else
271             return -1;
272     }

275     /**
276      * Returns SQL insert query that inserts row to <i>psa_log</i> database table.
277      *
278      * Override this method if you want to change log format in the database. You will have
279      * to manually alter log database table if you wish to add some new columns into it.
280      * This method has to have matching {@link format_database_log_query_params()} method.
281      *
282      * @param int $log_storage see {@link log()} method
283      * @see format_database_log_query_params()
284      * @see format_file_log()
285      * @see log()
286      * @return string
287      */
288     protected function format_database_log_query($log_storage){
289 
290         // log query. Should be formated for prepared query.
291         return "INSERT INTO " . Psa_Registry::get_instance()->PSA_CFG['logging']['storage'][$log_storage]['target'] . " (client_ip, log_time, request_uri, user_agent, referer, type, username, user_id, message, function, group_id, groupname) VALUES (?,NOW(),?,?,?,?,?,?,?,?,?,?)";
292     }

295     /**
296      * Returns array of parameters for prepared query from {@link format_database_log_query()} method.
297      *
298      * @param array $log_data see {@link log()} method
299      * @param int $log_storage see {@link log()} method
300      * @see format_database_log_query()
301      * @see format_file_log()
302      * @see log()
303      * @return array
304      */
305     protected function format_database_log_query_params($log_data){
306 
307         // parameters for prepared log query
308         return array(@$_SERVER["REMOTE_ADDR"],
309             @$_SERVER["REQUEST_URI"],
310             @$_SERVER["HTTP_USER_AGENT"],
311             @$_SERVER['HTTP_REFERER'],
312             @$log_data['type'],
313             @$log_data['username'],
314             @$log_data['user_id'],
315             @$log_data['message'],
316             @$log_data['function'],
317             @$log_data['group_id'],
318             @$log_data['groupname']);
319     }

322     /**
323      * Formats log message that will be written into the log file.
324      *
325      * Override this method if you want to change log format.
326      *
327      * @param array $log_data see {@link log()} method
328      * @param int $log_storage see {@link log()} method
329      * @see format_database_log()
330      * @see log()
331      * @return string
332      */
333     protected function format_file_log($log_data){
334 
335         // config array
336         $PSA_CFG = Psa_Registry::get_instance()->PSA_CFG;
337 
338         // format log message
339         if(@$PSA_CFG['logging']['more_lines']){
340             $new_line = "\r\n";
341             $message = "\r\n" . '[' . date($PSA_CFG['logging']['time_format']) . "] " . $new_line . '=====================' . $new_line;
342         }
343         else{
344             $new_line = '';
345             $message = '[' . date($PSA_CFG['logging']['time_format']) . "] " . $new_line;
346         }
347 
348         if(@$log_data['message'])
349             $message .= $log_data['message'] . $new_line;
350         if(@$_SERVER['REMOTE_ADDR'])
351             $message .= " IP={$_SERVER['REMOTE_ADDR']}" . $new_line;
352         if(@$log_data['username'])
353             $message .= " USER={$log_data['username']}" . $new_line;
354         if(@$log_data['user_id'])
355             $message .= " UID={$log_data['user_id']}" . $new_line;
356         if(@$log_data['grouname'])
357             $message .= " GROUP={$log_data['grouname']}" . $new_line;
358         if(@$log_data['group_id'])
359             $message .= " GID={$log_data['group_id']}" . $new_line;
360         if(@$log_data['function']){
361             if(!$new_line)
362                 $message .= ' ' . trim('FUNCTION=' . str_replace(array("\r\n","\n"), ' ', $log_data['function'])) . $new_line;
363             else
364                 $message .= ' ' . trim('FUNCTION=' .  $log_data['function']) . $new_line;
365         }
366         if(@$log_data['type'])
367             $message .= " TYPE={$log_data['type']}" . $new_line;
368         if(@$_SERVER['REQUEST_URI'])
369             $message .= " REQUEST_URI={$_SERVER['REQUEST_URI']}" . $new_line;
370         if(@$_SERVER['HTTP_USER_AGENT'])
371             $message .= " USER_AGENT={$_SERVER['HTTP_USER_AGENT']}" . $new_line;
372         if(@$_SERVER['HTTP_REFERER'])
373             $message .= " REFERER={$_SERVER['HTTP_REFERER']}" . $new_line;
374 
375         return $message . "\n";
376     }
