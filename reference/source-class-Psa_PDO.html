 32 /**
 33  * It's a simple database connection class and database access wrapper class.
 34  *
 35  * Handles database connection and executes queries against the database.
 36  * PSA uses {@link http://www.php.net/pdo PHP Data Objects (PDO)} extension for
 37  * database access. Methods in this class are wrappers for corresponding PDO methods.
 38  *
 39  * If you don't want to use methods from this class in your models, you can use
 40  * same database connection that can be accessed through <var>$this->psa_database->pdo</var>
 41  * which is PDO object.
 42  *
 43  * See in examples how to interact with database in model methods.
 44  *
 45  * <br><b>Examples:</b>
 46  *
 47  * <br><code>
 48  * class my_model extends Psa_Model {
 49  *
 50  *     // returns all records from a database table
 51  *     function get_all_records(){
 52  *
 53  *         $this->psa_database->query("SELECT * FROM table");
 54  *         return $this->psa_database->fetch_all();
 55  *     }
 56  *
 57  *
 58  *     // or the same as above
 59  *     function get_all_records1(){
 60  *         return $this->psa_database->fetch_all("SELECT * FROM table");
 61  *     }
 62  *
 63  *
 64  *     // returns one row from the table
 65  *     function get_one_row(){
 66  *         return $this->psa_database->fetch_row("SELECT * FROM table WHERE id = 123");
 67  *     }
 68  *
 69  *
 70  *     // iterates through each row
 71  *     function sum_all(){
 72  *
 73  *         $sum = 0;
 74  *
 75  *         $this->psa_database->query("SELECT * FROM table");
 76  *         while($row = $database->fetch_row()){
 77  *             $sum += $row['column1'] + $row['column2'] + $row['column3'];
 78  *         }
 79  *
 80  *         return $sum;
 81  *     }
 82  *
 83  *
 84  *     // executes prepared statement
 85  *     function save_values($value1, $value2, $value3){
 86  *
 87  *         // query
 88  *         $sql = "INSERT INTO table (column1, column2, column3) VALUES (?,?,?)";
 89  *
 90  *         // argument for execute() method
 91  *         $qparams = array($value1, $value2, $value3);
 92  *
 93  *         // execute query
 94  *         $this->psa_database->execute($qparams, $sql);
 95  *
 96  *         return $this->psa_database->last_insert_id();
 97  *     }
 98  *
 99  *
100  *     // use transactions
101  *     function save_values2($value1, $value2){
102  *
103  *         // start transaction
104  *         // be sure to have database connection before beginTransaction()
105  *         $this->psa_database->connect();
106  *         $this->psa_database->pdo->beginTransaction();
107  *
108  *         try{
109  *             $this->psa_database->execute(array($value1), "INSERT INTO table1 (column1) VALUES (?)");
110  *
111  *             $this->psa_database->execute(array($value2), "INSERT INTO table2 (column2) VALUES (?)");
112  *         }
113  *         catch (Psa_Db_Exception $e){
114  *             // rollback
115  *             $this->psa_database->pdo->rollback();
116  *         }
117  *
118  *         // commit
119  *         $this->psa_database->pdo->commit();
120  *     }
121  * }
122  * </code>
123  *
124  * @see config.php
125  */

177     /**
178      * Performs a query against the database.
179      *
180      * Throws a {@link Psa_Db_Exception} if query fails.
181      *
182      * @param string $sql SQL query.
183      * @return boolean True on success.
184      * @throws Psa_Db_Exception
185      */
186     public function query($sql){
187 
188         // connect to database if connection don't exists
189         if (!$this->pdo){
190             $this->connect();
191         }
192 
193         // destroy previous result
194         $this->result = null;
195 
196         // run query against the database
197         try{
198             $this->result = $this->pdo->query($sql);
199         }
200         catch(PDOException $e){
201 
202             $error = $this->pdo->errorInfo();
203 
204             throw new Psa_Db_Exception('SQL error: ' . $e->getMessage() . '. In query: ' . $sql, $e->getCode());
205         }
206 
207         // if success
208         return true;
209     }

238     /**
239      * Connects to the database.
240      *
241      * If called without arguments, it first uses connection parameters from object properties ({@link $dsn},
242      * {@link $username}, {@link $password}, {@link $driver_options}) and if
243      * {@link $dsn} value is not set, it takes connection parameters from <var>$PSA_CFG</var> config array
244      * (set in {@link config.php} or config_override.php).
245      * Throws a {@link Psa_Db_Exception} if the attempt to connect to the requested database fails.
246      *
247      * @param string $dsn PDO data source name.
248      * @param string $username Database user.
249      * @param string $password Database user's password.
250      * @param string $driver_options PDO driver-specific connection options.
251      * @param bool $throw_PDOException If true PDOException will be thrown instead Psa_Db_Exception.
252      * @return boolean
253      * @throws Psa_Db_Exception
254      * @see config.php
255      */
256     public function connect($dsn = null, $username = null, $password = null, $driver_options = null, $throw_PDOException = 0){
257 
258         // if connected to the database
259         if(!$this->pdo){
260 
261             $PSA_CFG = Psa_Registry::get_instance()->PSA_CFG;
262 
263             // if connection parameters are passed as method arguments
264             if($dsn){
265                 $this->dsn = $dsn;
266                 $this->username = $username;
267                 $this->password = $password;
268                 $this->driver_options = $driver_options;
269             }
270 
271             // if connection parameters are not set as object properties use default
272             else if(!$this->dsn){
273                 $this->dsn = $PSA_CFG['pdo']['dsn'];
274                 $this->username = $PSA_CFG['pdo']['username'];
275                 $this->password = $PSA_CFG['pdo']['password'];
276                 if(isset($PSA_CFG['pdo']['driver_options']))
277                     $this->driver_options = $PSA_CFG['pdo']['driver_options'];
278             }
279 
280             // Connect to database server
281             try{
282                 $this->pdo = new PDO($this->dsn, $this->username, $this->password, $this->driver_options);
283             }
284             catch(PDOException $e){
285 
286                 if($throw_PDOException)
287                     throw $e;
288 
289                 // check if PDO extension is enabled
290                 if(substr($this->dsn,0,6) == 'mysql:' && !extension_loaded('pdo_mysql'))
291                     $msg1 = "pdo_mysql extension not enabled.";
292                 else if(substr($this->dsn,0,6) == 'pgsql:' && !extension_loaded('pdo_pgsql'))
293                     $msg1 = "pdo_pgsql extension not enabled.";
294                 else
295                     $msg1 = '';
296 
297                 $message = "Unable to connect to database. $msg1 " . $e->getMessage();
298 
299                 // log message cannot be written into the database if there is problem with database connection
300                 if($PSA_CFG['logging']['storage']['psa_default']['type'] == 'database')
301                     throw new Psa_Db_Exception($message, $e->getCode(), false);
302                 else
303                     throw new Psa_Db_Exception($message, $e->getCode());
304             }
305         }
306 
307         // return true on success or already connected
308         return true;
309     }

396     /**
397      * Executes a prepared statement.
398      *
399      * Throws a {@link Psa_Db_Exception} if statement fails.
400      *
401      * @param array $params An array of values with as many elements as there are bound
402      * parameters in the SQL statement being executed.
403      * @param string|PDOStatement $statement SQL query or prepared PDO statement.
404      * @param bool $throw_PDOException If true PDOException will be thrown instead Psa_Db_Exception
405      * @return bool
406      * @see prepare()
407      */
408     public function execute($params = null, $statement = null, $throw_PDOException = 0) {
409 
410         // if statement is not set
411         if(!$statement)
412             $statement = $this->result;
413         else if($statement instanceof PDOStatement)
414             $this->result = $statement;
415         // if $statement is SQL query
416         else
417             $this->result = $statement = $this->prepare($statement);
418 
419         // execute prepared query
420         try{
421             if(is_array($params))
422                 $statement->execute($params);
423             else
424                 $statement->execute();
425         }
426         catch(PDOException $e){
427 
428             if($throw_PDOException)
429                 throw $e;
430 
431             $error = $this->pdo->errorInfo();
432 
433             if(is_array($params))
434                 $query_params_str = ' Query input parameters: ' . implode(',',$params);
435             else
436                 $query_params_str = '';
437 
438             throw new Psa_Db_Exception('SQL error: ' . $e->getMessage() . '. In query: ' . $statement->queryString . $query_params_str, $e->getCode());
439         }
440 
441         // if success
442         return true;
443     }
