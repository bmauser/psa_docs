32 /**
33  * Class for an object that can be used for storing results from model methods.
34  *
35  * The only purpose of <kbd>Psa_Result</kbd> object is to transfer results from model to view, 
36  * but a nicer way to do that is to make model methods to return the value and pass it 
37  * to the view method as argument in the controller action method.
38  * In <i>models</i> and <i>views</i> you can access <kbd>Psa_Result</kbd> object through
39  * references in local scope.
40  * For example, you can add property to <kbd>Psa_Result</kbd> object in your model
41  * by
42  * <code>
43  * $this->psa_result->my_value = '123';
44  * </code>
45  *
46  * Or you can get reference to <kbd>Psa_Result</kbd> object from any scope with:
47  * <code>
48  * Psa_Result::get_instance();
49  * </code>
50  *
51  * It implements {@link http://en.wikipedia.org/wiki/Singleton_pattern singleton pattern}
52  * so you can get reference to result object from any scope with
53  * {@link get_instance()} method. You cannot make instance of Psa_Result object with
54  * the <var>new</var> operator.
55  *
56  * <b>Example:</b>
57  *
58  * <code>
59  * <?php
60  *
61  * class example_model extends Psa_Model {
62  *
63  *  function get_numbers(){
64  *
65  *      // put some numbers in the result object using the reference in local scope
66  *      $this->psa_result->numbers = array(1,2,3);
67  *
68  *      // or the same thing but using get_instance() method. This works from any scope.
69  *      Psa_Result::get_instance()->more_numbers = array(4,5,6);
70  *  }
71  * }
72  *
73  * ?>
74  * </code>
75  */
76 class Psa_Result extends Psa_Singleton {
77 
78     /**
79      * Returns object's instance.
80      *
81      * You should statically call this method with scope resolution operator
82      * (<kbd>Psa_Result::get_instance()</kbd>) which gives you
83      * instance to the object from any scope in your application.
84      *
85      * @return Psa_Result
86      */
87     public static function get_instance($classname = __CLASS__){
88         return parent::get_instance($classname);
89     }
90 
91 }
