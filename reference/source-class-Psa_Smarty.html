51 
52     function __construct($template_dir = null){
53 
54         $PSA_CFG = Psa_Registry::get_instance()->PSA_CFG;
55 
56         // call parent constructor
57         parent::__construct();
58 
59         // Smarty folder configuration
60         $this->cache_dir    = PSA_BASE_DIR . '/' . $PSA_CFG['folders']['smarty']['cache_dir'];
61         $this->config_dir   = PSA_BASE_DIR . '/' . $PSA_CFG['folders']['smarty']['config_dir'];
62         $this->compile_dir  = PSA_BASE_DIR . '/' . $PSA_CFG['folders']['smarty']['compile_dir'];
63         if(!$template_dir)
64             $template_dir = $PSA_CFG['folders']['smarty']['template_dir'];
65         $this->template_dir = PSA_BASE_DIR . '/' . $template_dir;
66 
67         $this->use_sub_dirs = true;
68 
69         // suppress notices by default like in smarty 2
70         $this->error_reporting = error_reporting() & ~E_NOTICE;
71 
72         // in develop mode
73         if($PSA_CFG['develop_mode']){
74             $this->force_compile = true;
75         }
76     }
77 }
