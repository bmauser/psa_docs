 32 /**
 33  * Class for views that use {@link http://www.smarty.net Smarty} template engine.
 34  *
 35  * Classes that extend {@link Psa_Dully_View} or {@link Psa_Smarty_View} class are the place for methods that will
 36  * print results from your application. This can be HTML, XML or something else like a plain text.
 37  * View methods, like model methods, are called from the controller. All application logic should
 38  * be in model, and in view you should put only part of the logic for displaying results (presentation logic) like
 39  * template rendering code.
 40  *
 41  * You can write <i>Psa_Smarty_View</i> views by extending this class. Here is an example:
 42  *
 43  * <code>
 44  * <?php
 45  * class example_view extends Psa_Smarty_View{
 46  *
 47  *     // Render even_numbers.tpl template and put it in "result" smarty value
 48  *     function even_numbers(){
 49  *         $rendered_result = $this->psa_smarty->fetch('even_numbers.tpl');
 50  *         $this->psa_smarty->assign("result", $rendered_result);
 51  *     }
 52  *
 53  *     // Main layout render and display method.
 54  *     function generate_html(){
 55  *
 56  *         $rendered_menu = $this->psa_smarty->fetch('menu.tpl');
 57  *         $rendered_header = $this->psa_smarty->fetch('header.tpl');
 58  *
 59  *         $this->psa_smarty->assign('header', $rendered_header);
 60  *         $this->psa_smarty->assign('menu', $rendered_menu);
 61  *
 62  *         echo $this->psa_smarty->fetch('example_app/main.tpl');
 63  *     }
 64  * }
 65  * ?>
 66  * </code>
 67  *
 68  * @see Psa_Dully_View
 69  * @see Psa_Smarty
 70  */
 71 class Psa_Smarty_View extends Psa_View{
 72 
 73 
 74     /**
 75      * Reference to Smarty template engine object.
 76      * @var Psa_Smarty
 77      */
 78     public $psa_smarty;
 79 
 80 
 81 
 82     /**
 83      * Constructor.
 84      */
 85     function __construct(){
 86 
 87         parent::__construct();
 88 
 89         // set reference to $psa_smarty object if exists
 90         if(!$this->psa_registry->psa_smarty instanceof Psa_Smarty){
 91 
 92             // include smarty class file
 93             include_once PSA_BASE_DIR . '/lib/Psa_Smarty.php';
 94 
 95             $this->psa_registry->psa_smarty = new Psa_Smarty;
 96 
 97             // assign result object to smarty
 98             $this->psa_registry->psa_smarty->assign('psa_result', $this->psa_result);
 99             $this->psa_registry->psa_smarty->assign('r', $this->psa_result);
100         }
101 
102         $this->psa_smarty = $this->psa_registry->psa_smarty;
103     }
104 }
