 32 /**
 33  * User object class.
 34  *
 35  * <br><b>Examples:</b>
 36  *
 37  * <br><b>1)</b> Create a new user:
 38  * <code>
 39  * // user object
 40  * $user = new Psa_User('new');
 41  *
 42  * // set username and password
 43  * $user->username = 'my_user';
 44  * $user->password = 'user_password';
 45  *
 46  * // save user to the database
 47  * $user->save();
 48  * </code>
 49  *
 50  * <br><b>2)</b> Authorize user by username and password:
 51  * <code>
 52  * $user = new Psa_User('my_user');
 53  * $user->authorize('user_password');
 54  * </code>
 55  *
 56  * <br><b>3)</b> Extend <kbd>Psa_User</kbd> class. You'll probably want to extend <kbd>Psa_User</kbd> class
 57  * and to add more columns to <i>psa_user</i> database table. In this example <kbd>user_level</kbd>,
 58  * <kbd>photo</kbd> and <kbd>email</kbd> columns are added to <i>psa_user</i> table.
 59  *
 60  * <code>
 61  * class MyUser extends Psa_User{
 62  *
 63  *     public function __construct($user_id_or_username){
 64  *         parent::__construct($user_id_or_username, array('id', 'username', 'user_level', 'photo', 'email'));
 65  *     }
 66  * }
 67  * </code>
 68  *
 69  * Code in a model method:
 70  * <code>
 71  * // create a new user
 72  * $user = new MyUser('new');
 73  * $user->username = 'my_user';
 74  * $user->user_level = 'admin';
 75  * $user->email = 'admin@host';
 76  * $user->save();
 77  * </code>
 78  *
 79  * <br><b>4)</b> Change user's group membership:
 80  * <code>
 81  * // user object
 82  * $user = new MyUser(10); // MyUser extends Psa_User
 83  *
 84  * // remove the user from all groups and put it in groups with ID 3,5 and 7
 85  * $user->remove_group('all');
 86  * $user->add_group(array(3,5,7));
 87  * </code>
 88  */

131     /**
132      * Constructor method.
133      *
134      * - If called with '<kbd>new</kbd>' as the first argument, new user will be created when {@link save()}
135      *   method is called.
136      * - If called with user ID as the first argument, {@link $id} value is set. In this case, the
137      *   first argument must be an integer value.
138      * - If the first parameter is a string (different from '<kbd>new</kbd>'), the {@link $username} value is set.
139      *
140      * @param int|string $user_id_or_username '<kbd>new</kbd>' if new user should be created, or ID or username of the existing user.
141      * @param array $table_columns Array with column names in <i>psa_user</i> database table. See example 3
142      * above on how to extend this class and add more columns to the database table.
143      * @see restore()
144      */
145     public function __construct($user_id_or_username, array $table_columns = array('id', 'username')){
146 
147         if(!$user_id_or_username)
148             throw new Psa_User_Exception('Invalid user ID or username.', 201);
149 
150         // new user
151         if($user_id_or_username === 'new'){
152             parent::__construct('psa_user', 'id', null, $table_columns, 'psa_user_id_seq');
153         }
154         // existing user
155         else{
156 
157             // is int
158             if(psa_is_int($user_id_or_username)){
159                 $this->id = (int) $user_id_or_username;
160                 parent::__construct('psa_user', 'id', $this->id, $table_columns, 'psa_user_id_seq');
161             }
162             else{
163 
164                 if(!is_string($user_id_or_username))
165                     throw new Psa_User_Exception('Invalid username.', 202);
166 
167                 $this->username = $user_id_or_username;
168                 parent::__construct('psa_user', 'username', $this->username, $table_columns, 'psa_user_id_seq');
169             }
170         }
171     }

174     /**
175      * Does the same as {@link restore()} method, but also writes a log message and calls
176      * {@link session_save()} method.
177      *
178      * See examples in {@link Psa_User} class description.
179      *
180      * This function calls {@link restore()} method with <kbd>'username_password'</kbd> as the first argument
181      * and writes a log message that user is authorized.
182      *
183      * @param string $password User password.
184      * @param bool $write_success_login_logs If false, log message about successful authorization will not be written.
185      * @param bool $save_to_session If true, {@link session_save()} will be called to write user ID and username to
186      * the session if session is started before.
187      * @return int 1 for success.
188      * @see restore()
189      * @throws Psa_User_Exception
190      */
191     public function authorize($password = null, $write_success_login_logs = true, $save_to_session = true){
192 
193         if($password){
194             $this->password = $this->password_hash($password);
195             $return = $this->restore('username_password');
196         }
197         else{
198             $return = $this->restore();
199         }
200 
201         // write log
202         if($write_success_login_logs)
203             $this->log('User authorized', __METHOD__, 2);
204 
205         if($save_to_session)
206             $this->session_save(false);
207 
208         return $return;
209     }

212     /**
213      * Restores member variables from the database previously saved with {@link save()} method.
214      *
215      * Reads values from the database and populates corresponding object member variables. For example,
216      * if you have added to <i>psa_user</i> table column named <kbd>access_level</kbd> and you passed
217      * the name of the column to the constructor with <var>$table_columns</var> parameter, then when
218      * this method is called, member variable with name <kbd>access_level</kbd> will be
219      * populated with the value from the database.
220      *
221      * @see authorize()
222      * @throws Psa_User_Exception
223      */
224     public function restore($restore_by = null, $try_from_session = false){
225 
226         try{
227             if($restore_by == 'username_password'){
228                 if(!$this->username or !$this->password)
229                     throw new Psa_User_Exception("Username and password not set.", 203);
230 
231                 $sql = "SELECT <COLUMNS> FROM {$this->psa_registry->PSA_CFG['database']['table']['user']} WHERE username=? AND password=?";
232                 $q_params = array($this->username, $this->password);
233                 $this->restore_from_database(array(), $sql, $q_params);
234             }
235             else if($this->id or $this->username){
236 
237                 if($try_from_session){
238                     try{
239                         $this->restore_from_session();
240                     }
241                     catch (Psa_Active_Record_Exception $e){
242                         $this->restore_from_database();
243                     }
244                 }
245                 else
246                     $this->restore_from_database();
247             }
248             else
249                 throw new Psa_User_Exception('Cannot restore user data. User id or username not set', 204);
250         }
251         catch (Psa_Active_Record_Exception $e){
252             throw new Psa_User_Exception('Cannot restore user data. ' . $e->getMessage(), 205);
253         }
254     }

257     /**
258      * Saves values from the member variables to the database.
259      *
260      * There must be a column in the database named the same as the member variable. It works only for
261      * variables mentioned in <var>$table_columns</var> argument to the constructor or by <var>$only_columns</var>
262      * argument to this method.
263      * All saved values will be restored when the {@link authorize()} or {@link restore()}
264      * methods are called.
265      * See examples in {@link Psa_User} class description.
266      *
267      * @see authorize()
268      * @param array $only_columns Array with column names in <i>psa_user</i> database table to work with.
269      * If not set, column names set by the constructor are used.
270      * @return int 1 for success
271      * @throws Psa_User_Exception
272      */
273     public function save(array $only_columns = array()){
274 
275         if($this->psa_new_record && !($this->username && $this->password))
276             throw new Psa_User_Exception("Error creating a new user. Username & password not set.", 206);
277 
278 
279         if($this->psa_new_record){
280 
281             if(!isset($this->password) or !$this->password)
282                 throw new Psa_User_Exception("Password for new user not set.", 214);
283 
284             // hash new password
285             $this->password = $this->password_hash($this->password);
286 
287             parent::save_to_database($only_columns, array('password'));
288 
289             // write log
290             $this->log('New user created', __METHOD__, 2);
291         }
292         else{
293             parent::save_to_database($only_columns);
294         }
295 
296         return $this->id;
297     }

357     /**
358      * Puts user in the group or removes user from group (or more groups if $group_id argument is array).
359      *
360      * This method is called from add_group() and remove_group() methods.
361      *
362      * @param int|array|string $group_id id of the group or array with group ids.
363      * @param int $action 1 add, 0 remove
364      * @return int 1 for success, -1
365      * @see add_group()
366      * @see remove_group()
367      * @throws Psa_User_Exception
368      * @ignore
369      */
370     protected function add_remove_group($group_id, $action){
371 
372         if(!$this->id)
373             throw new Psa_User_Exception('Error changing user group. User ID not set.', 207);
374 
375         if(!$group_id)
376             throw new Psa_User_Exception('Error changing user group. Invalid group ID.', 208);
377 
378 
379         // if $group_id is not array make it array for foreach loop
380         if(!is_array($group_id))
381             $group_id = array($group_id);
382 
383         // flag if some query failed in the foreach loop
384         $success = $failed = 0;
385 
386         // add user to one or more groups or remove from group(s)
387         foreach ($group_id as $group_id_key => $group_id_value){
388 
389             // add to group
390             if($action == 1){
391                 $sql = "INSERT INTO {$this->psa_registry->PSA_CFG['database']['table']['user_in_group']} (user_id, group_id) VALUES (?, ?)";
392                 $q_params = array($this->id, $group_id_value);
393             }
394             // remove from group
395             else if($action == 0){
396 
397                 // remove all groups
398                 if($group_id_value == 'all'){
399                     $sql = "DELETE FROM {$this->psa_registry->PSA_CFG['database']['table']['user_in_group']} WHERE user_id=?";
400                     $q_params = array($this->id);
401                 }
402                 // remove specific group
403                 else{
404                     $sql = "DELETE FROM {$this->psa_registry->PSA_CFG['database']['table']['user_in_group']} WHERE user_id=? AND group_id=?";
405                     $q_params = array($this->id, $group_id_value);
406                 }
407             }
408 
409             // run query against the database
410             $this->psa_database->execute($q_params, $sql);
411 
412             // if no rows affected user already wasn't or was in the group depending on $action
413             if($this->psa_database->affected_rows() <= 0)
414                 $failed = 1;
415             else
416                 $success = 1;
417         }
418 
419         if($success){
420 
421             // write log
422             if($this->psa_registry->PSA_CFG['logging']['max_log_level'] >= 2)
423                 $this->log("Group membership changed: " . implode(',',$group_id) . " action=" . ($action ? 'add' : 'remove'),__METHOD__,2);
424 
425             if(!$failed)
426                 return 1;
427         }
428 
429         return -1;
430     }

433     /**
434      * Changes user password.
435      *
436      * New password will be hashed with {@link password_hash()} method and stored in the database.
437      * You need to call this method only when you want to change the password for the existing user.
438      * Throws {@link Psa_User_Exception} on error.
439      *
440      * @param string $new_password New password.
441      * @return int 1 for success.
442      * @throws Psa_User_Exception
443      * @see password_hash()
444      */
445     public function password_change($new_password){
446 
447         if((string)$new_password !== $new_password)
448             throw new Psa_User_Exception('Error changing password. Invalid password.', 209);
449 
450         if($new_password){
451 
452             // encrypt new password
453             $new_password = $this->password_hash($new_password);
454 
455             // update user in the database
456             if($this->id){
457                 $sql = "UPDATE {$this->psa_registry->PSA_CFG['database']['table']['user']} SET password = ? WHERE id = ?";
458                 $q_params = array($new_password, $this->id);
459             }
460             else{
461                 $sql = "UPDATE {$this->psa_registry->PSA_CFG['database']['table']['user']} SET password = ? WHERE username = ?";
462                 $q_params = array($new_password, $this->username);
463             }
464 
465             // run query against the database
466             try{
467                 $this->psa_database->execute($q_params, $sql);
468 
469                 // write log
470                 $this->log("Password changed",__METHOD__,2);
471 
472                 return 1;
473             }
474             catch (Psa_Db_Exception $e){
475                 throw new Psa_User_Exception('Error changing password', 210);
476             }
477         }
478     }

481     /**
482      * Verifies user password.
483      *
484      * Checks if given string is the user's password. This method can be used in process of changing user password to
485      * verify the old password.
486      *
487      * @param string $password Password to check.
488      * @return int 1 given password is valid, 0 given password is invalid
489      * @see password_hash()
490      * @see password_change()
491      * @throws Psa_User_Exception
492      */
493     public function password_verify($password){
494 
495         if(!$password)
496             throw new Psa_User_Exception('Error verifying password.', 211);
497 
498 
499         // serialize object
500         $database_password = $this->password_hash($password);
501 
502         // update user in the database
503         if($this->id){
504             $sql = "SELECT id FROM {$this->psa_registry->PSA_CFG['database']['table']['user']} WHERE password = ? AND id = ?";
505             $q_params = array($database_password, $this->id);
506         }
507         else{
508             $sql = "SELECT id FROM {$this->psa_registry->PSA_CFG['database']['table']['user']} WHERE password = ? AND username = ?";
509             $q_params = array($database_password, $this->username);
510         }
511 
512         $this->psa_database->execute($q_params, $sql);
513 
514         $row = $this->psa_database->fetch_row();
515 
516         if($row['id'])
517             return 1;
518         else
519             return 0;
520     }

570     /**
571      * Returns an array with all user's groups.
572      *
573      * Example of returning array:
574      * <pre>
575      * Array
576      * (
577      *     [11] => 'admins'  // array index is group id and value is group name
578      *     [14] => 'users'
579      *     [33] => 'other'
580      * )
581      * </pre>
582      *
583      * {@link $id} member variable must be set before calling this method. If you pass username
584      * to the constructor, call {@link restore()} or {@link authorize()} method before to get the

586      *
587      * @return array

589     public function get_groups(){
590 
591         if(!$this->id)
592             throw new Psa_User_Exception('Error with setting user groups. User ID not set.', 212);
593 
594 
595         // get data from all groups user is in
596         $sql = "SELECT psa_user_in_group.group_id, psa_group.name FROM psa_user_in_group JOIN psa_group ON psa_user_in_group.group_id = psa_group.id WHERE psa_user_in_group.user_id = ?";
597         $q_params = array($this->id);
598 
599         $this->psa_database->execute($q_params, $sql);
600 
601         $groups = array();
602 
603         // for each fetched row

605             $groups[$row['group_id']] = $row['name'];
606         }
607 
608         return $groups;

632     /**
633      * Stores user ID and username into the session.
634      *
635      * It will store an array named <kbd>psa_current_user_data</kbd> with user ID and username into the session.
636      * All log messages will contain user ID and username from this session
637      * variable if not explicitly set otherwise. See {@link Psa_Logger::log()} method for details.
638      *
639      * @param bool $throw_exception If true, Psa_User_Exception will be thrown if PHP session is not started.
640      * @throws Psa_User_Exception
641      */
642     protected function session_save($throw_exception = true){
643 
644         // check if session is started
645         if(!session_id()){
646             if($throw_exception)
647                 throw new Psa_User_Exception('Session is not started. Cannot store user data into the session.', 213);
648             else
649                 return false;
650         }
651 
652         if($this->id)
653             $_SESSION['psa_current_user_data']['id'] = $this->id;
654 
655         if($this->username)
656             $_SESSION['psa_current_user_data']['username'] = $this->username;
657 
658         $this->save_to_session();
659 
660         return true;
661     }
