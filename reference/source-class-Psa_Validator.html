  32 /**
  33  * Class for input data validation.
  34  *
  35  * You should validate all input data from users.
  36  * Validation should occur in the model method, not in the controller (in some action method).
  37  * That makes your models more independent and more reusable and testing is easier.
  38  *
  39  * <br><b>Usage example</b>
  40  *
  41  * <code>
  42  * <?php
  43  *
  44  * // model class
  45  * class search extends Psa_Model{
  46  *
  47  *     // search method
  48  *     function filter($max_results, $only_pages){
  49  *
  50  *         // input validation
  51  *         $validator = new Psa_Validator();
  52  *         $validator->required($max_results, 'int');
  53  *         $validator->optional($only_pages, 'int_array');
  54  *
  55  *         // Method logic starts here.
  56  *         // For example to query the database using $max_results in SQL query
  57  *         // ...
  58  * //
  59  *     }
  60  * }
  61  * ?>
  62  * </code>
  63  *
  64  * <br><b>All validation actions</b>
  65  *
  66  * Two main methods in this class are {@link required()} and {@link optional()}. They behave the same way except that
  67  * {@link optional()} will not raise exception if value is empty or <kbd>null</kbd>.
  68  * Like in the example above, first argument is the value that should be validated and second is the name of the 
  69  * validation action. See {@link required()} method documentation for all action names.
  70  * Some validation actions require more parameters. Examples:
  71  *
  72  * <code>
  73  *
  74  * $val = new Psa_Validator();
  75  *
  76  * // Finds whether the given $value is between (or equal) 10 and 20.
  77  * $val->required($value, 'between', 10, 20);
  78  *
  79  * // Finds whether the length of the given $value is between (or equal) 10 and 20 characters.
  80  * $val->required($value, 'lenbetween', 10, 20);
  81  *
  82  * // Finds whether the given $value is in array passed as third argument.
  83  * $val->required($value, 'invalues', array(1, 2, 3, 4));
  84  *
  85  * // Checks if the given $value is an integer.
  86  * $val->required($value, 'int');
  87  *
  88  * // Finds whether the given $value is an integer bigger than 0.
  89  * $val->required($value, 'id');
  90  *
  91  * // Checks date if matches given format and validity of the date.
  92  * // Third argument is combination of 'mm', 'dd' and 'yyyy'
  93  * $val->required($value, 'date', 'mm.dd.yyyy');
  94  *
  95  * //  Finds whether the given $value has valid email format.
  96  * $val->required($value, 'email');
  97  *
  98  * // Checks is the given $value a valid IPv4 address.
  99  * $val->required($value, 'ip4');
 100  *
 101  * // Checks is the given $value a valid floating point number.
 102  * $val->required($value, 'float');
 103  *
 104  * // Checks whether the given $value is a valid string.
 105  * $val->required($value, 'string');
 106  *
 107  * // Checks whether the given $value contains letters only.
 108  * $val->required($value, 'alpha');
 109  *
 110  * // Checks whether the given $value contains numbers only.
 111  * $val->required($value, 'num');
 112  *
 113  * // Checks whether the given $value contains alphanumeric characters only.
 114  * $val->required($value, 'alphanum');
 115  *
 116  * // Checks whether the given $value contains domain name safe characters only.
 117  * $val->required($value, 'domainsafe');
 118  *
 119  * // The same as 'domainsafe' but dot '.' is also included.
 120  * $val->required($value, 'hostname');
 121  *
 122  * // Searches $value for a match to the regular expression given as third argument
 123  * $val->required($value, 'regex', '/[0-9]/');
 124  *
 125  * // Checks if given $value is valid URL.
 126  * $val->required($value, 'url');
 127  *
 128  * // Calls user function or method which has to return true or false.
 129  * $val->required($value, 'callback', 'user_function_name', 'param1');
 130  *
 131  * // Returns true if parameters are equal (==).
 132  * $val->required($value, 'equal', 'something');
 133  *
 134  * // Returns true if parameters are identical (===).
 135  * $val->required($value, 'identical', 1234);
 136  *
 137  * // Validates an IPv6 address.
 138  * $val->required($value, 'ip6');
 139  *
 140  * </code>
 141  *
 142  * <br><b>Validating array of values</b>
 143  *
 144  * If you want to validate an array of values, suffix <b><kbd>_array</kbd></b> can be added to any of
 145  * validation actions to validate each element of array. For example, this is useful
 146  * when validating input from HTML drop-down menu with multiple selections.
 147  *
 148  * <code>
 149  *
 150  * $val = new Psa_Validator();
 151  *
 152  * // each element of $array should be integger
 153  * $val->required($array, 'int_array');
 154  *
 155  * // each element of $array should be 1, 2, 3 or 4
 156  * $val->required($array, 'invalues_array', array(1, 2, 3, 4));
 157  *
 158  * </code>
 159  *
 160  * <br><b>Custom error message per validation</b>
 161  *
 162  * If validation fails, by default exception will be raized. Each action has its own default
 163  * error message, but you can change that message by passing one more argument to {@link required()}
 164  * or {@link optional()} method. This can be usefull if you want more descriptive error message
 165  * or message will be shown to the user.
 166  *
 167  * <code>
 168  *
 169  * $val = new Psa_Validator();
 170  *
 171  * $val->required($value, 'id', 'Not valid product ID');
 172  *
 173  * </code>
 174  *
 175  * <br><b>Adding custom validation</b>
 176  *
 177  * You can extend this class and add your custom validation function (action). In the next example '<kbd>5to9</kbd>'
 178  * action is added. Note that validation method named <var>check_5to9()</var> and error message named
 179  * <var>$msg_check_5to9</var> are added in the child class. Validation method should return <kbd>true</kbd> or
 180  * <kbd>false</kbd>. When you add validation action like this you can use it with <kbd>_array</kbd> suffilx like
 181  * any other built-in action. See the source code of this class for more examples.
 182  *
 183  * <code>
 184  * class MyValidator extends Psa_validator{
 185  *
 186  *     // Error message for 5to9 validation
 187  *     // %v will be replaced with value to validate, and %p1, %p2...
 188  *     // will be replaced with other parameters to validation method.
 189  *     $msg_check_5to9 = "%v does not contains only digits from 5 to 9";
 190  *
 191  *
 192  *     // Returns true if $value contains only digits from 5 to 9
 193  *     public function check_5to9($value){
 194  *
 195  *         if(preg_match('/^[5-9]*$/', $value))
 196  *             return true;
 197  *         return false;
 198  *     }
 199  * }
 200  * </code>
 201  *
 202  *
 203  */

 458     /**
 459      * Validates values set by {@link required()} and {@link optional()} methods.
 460      * If validation of any value fails, raises exception (or returns false, see {@link __construct()}),
 461      * otherwise returns true. See example in {@link Psa_Validator class description}.
 462      * It will throw {@link Psa_Validation_Exception} when validation fails. For example, you can
 463      * catch this exception in controller and call another action.
 464      *
 465      * @see required()
 466      * @see optional()
 467      * @return bool
 468      * @throws Psa_Validation_Exception
 469      * @ignore
 470      */
 471     protected function validate($reqired, $params){
 472 
 473         $value_to_validate = $params[0];
 474         $validation_type = $params[1];
 475 
 476         // check if value can be empty
 477         if($this->is_empty($value_to_validate)){
 478             if(!$reqired)
 479                 return true;
 480 
 481             return $this->fail_local("value is required for '{$validation_type}' validation", $value_to_validate, $validation_type, 601);
 482         }
 483 
 484         // if validation has to be run on array of elements
 485         if(substr($validation_type, -6) == '_array'){
 486 
 487             if(is_array($value_to_validate)){
 488 
 489                 // find type of validation
 490                 $validation_type = substr($validation_type, 0, -6);
 491 
 492                 $return = true;
 493 
 494                 // validate each element
 495                 foreach ($value_to_validate as $data_key => $data_value){
 496 
 497                     // check if value can be empty
 498                     if($this->is_empty($data_value)){
 499                         // if $data can be empty (optional)
 500                         if(!$reqired)
 501                             continue;
 502                         // data is required
 503                         else
 504                             return $this->fail_local("value is required for '{$validation_type}' validation", $data_value, $validation_type, 602);
 505                     }
 506                     $params[0] = $data_value;
 507                     $params[1] = $validation_type;
 508                     if(!$this->invoke_validation_method($params))
 509                         $return = false;
 510                 }
 511 
 512                 return $return;
 513             }
 514             else{
 515                 return $this->fail_local("'{$value_to_validate}' is not an array thus cannot validate each element", $value_to_validate, $validation_type, 603);
 516             }
 517         }
 518         // single value
 519         else{
 520             return $this->invoke_validation_method($params);
 521         }
 522     }

 525     /**
 526      * Invokes validation functions and puts error messages into $this->errors array
 527      *
 528      * @ignore
 529      */
 530     protected function invoke_validation_method($params){
 531 
 532         $value_to_validate = $params[0];
 533         $validation_type = $params[1];
 534 
 535         // name of the method to invoke
 536         $validation_method_name = 'check_' . $validation_type;
 537 
 538         // if method exists
 539         if(method_exists($this, $validation_method_name)){
 540 
 541             // method reflection
 542             $invoke_method = new ReflectionMethod($this, $validation_method_name);
 543 
 544             // remove second element because it is validation_type
 545             array_splice($params, 1, 1);
 546 
 547             // call validating method
 548             $validation_result = $invoke_method->invokeArgs($this, $params);
 549 
 550             // call validating method
 551             if(!$validation_result){
 552 
 553                 // how many arguments has validation_method
 554                 $invoke_method_num_args = $invoke_method->getNumberOfParameters();
 555 
 556                 // default validation message
 557                 $message_var_name = 'msg_' . $validation_method_name;
 558 
 559                 // choose error message
 560                 if(isset($params[$invoke_method_num_args]) && is_string($params[$invoke_method_num_args]))
 561                     $message = $params[$invoke_method_num_args];
 562                 else if(isset($this->$message_var_name) && $this->$message_var_name)
 563                     $message = $this->$message_var_name;
 564                 else
 565                     $message = $this->msg_default;
 566 
 567                 if(count($params) > 1 && substr_count($message, '%p')){
 568                     // replace %p1, %p2, %p3... in message with function params
 569                     for($i=1; $i<$invoke_method_num_args; $i++){
 570                         if(is_array($params[$i]))
 571                             $val = '[' . implode(', ', $params[$i]) . ']';
 572                         else
 573                             $val = $params[$i];
 574                         $message = str_replace('%p' . $i, $val, $message);
 575                     }
 576                 }
 577 
 578                 // replace %v in message with value
 579                 $message = str_replace('%v', is_array($value_to_validate) ? 'Array' : $value_to_validate, $message);
 580 
 581                 return $this->fail_local($message, $value_to_validate, $validation_type, 604);
 582             }
 583 
 584             return true;
 585         }
 586         else
 587             return $this->fail_local("undefined validation method check_{$validation_type}()", $value_to_validate, $validation_type, 605);
 588     }

 591     /**
 592      * Finds whether the given <var>$value</var> is between (or equal) $min and $max.
 593      *
 594      * @param float $value the variable being evaluated.
 595      * @param float|bool $min if null only <kbd>$max</kbd> argument is evaluated
 596      * @param float|bool $max if null only <kbd>$min</kbd> argument is evaluated
 597      * @return bool
 598      * @ignore
 599      */
 600     public function check_between($value, $min = null, $max = null){
 601 
 602         if($value === '')
 603             return false;
 604 
 605         if($min !== null && $max !== null && $value >= $min && $value <= $max)
 606             return true;
 607         else if($min === null && $max !== null && $value <= $max)
 608             return true;
 609         else if($min !== null && $max === null && $value >= $min)
 610             return true;
 611 
 612         return false;
 613     }

 690     /**
 691      * Checks date if matches given format and validity of the date.
 692      *
 693      * Examples:
 694      *
 695      * <code>
 696      * Psa_Validator::check_date('22.22.2222', 'mm.dd.yyyy'); // returns false
 697      * Psa_Validator::check_date('11/30/2008', 'mm/dd/yyyy'); // returns true
 698      * Psa_Validator::check_date('30-01-2008', 'dd-mm-yyyy'); // returns true
 699      * Psa_Validator::check_date('2008 01 30', 'yyyy mm dd'); // returns true
 700      * </code>
 701      *
 702      * @param string $value the variable being evaluated.
 703      * @param string $format Format of the date. Any combination of <i>mm</i>, <i>dd</i>, <i>yyyy</i>
 704      * with a single character separator between.
 705      * @return bool
 706      * @ignore
 707      */
 708     public function check_date($value, $format = 'mm.dd.yyyy'){
 709 
 710         if(strlen($value) == 10 && strlen($format) == 10){
 711 
 712             // find separator. Remove all other characters from $format
 713             $separator_only = str_replace(array('m','d','y'),'', $format);
 714             $separator = $separator_only[0]; // separator is first character
 715 
 716             if($separator && strlen($separator_only) == 2){
 717                 // make regex
 718                 $regexp = str_replace($separator, "\\" . $separator, $format);
 719                 $regexp = str_replace('mm', '[0-1][0-9]', $regexp);
 720                 $regexp = str_replace('dd', '[0-3][0-9]', $regexp);
 721                 $regexp = str_replace('yyyy', '[0-9]{4}', $regexp);
 722 
 723                 if($regexp != $value && preg_match('/'.$regexp.'/', $value)){
 724 
 725                     // check date
 726                     $day   = substr($value, strpos($format, 'd'), 2);
 727                     $month = substr($value, strpos($format, 'm'), 2);
 728                     $year  = substr($value, strpos($format, 'y'), 4);
 729 
 730                     if(@checkdate($month, $day, $year))
 731                         return true;
 732                 }
 733             }
 734         }
 735         return false;
 736     }
