 32 /**
 33  * Deletes a user or a group. Removes all data about the user or group from the database.
 34  * Do not use this function directly. Use psa_delete_group() and psa_delete_user() wrapper functions.
 35  *
 36  * @param int|array|string $id id of the user|group or array with user|group ids. 'all' for
 37  * deleting all users|groups.
 38  * @param int $user_or_group what to delete. 1-user, 2-group
 39  * @return int 0 for failure, 1 for success, -1 user|group (or more users|groups) not exists
 40  * @see psa_delete_group()
 41  * @see psa_delete_user()
 42  * @ignore
 43  */
 44 function psa_del_user_group($id, $user_or_group){
 45 
 46     if($id && $user_or_group){
 47 
 48         // config array
 49         $PSA_CFG = Psa_Registry::get_instance()->PSA_CFG;
 50 
 51         // database object
 52         $psa_database = Psa_Registry::get_instance()->psa_database;
 53 
 54         // if $id is not array make it array for foreach loop
 55         if(!is_array($id))
 56             $id = array($id);
 57 
 58         // flag if some query fail in the foreach loop
 59         $failed = 0;
 60 
 61         // work with users
 62         if($user_or_group == 1){
 63             $table = $PSA_CFG['database']['table']['user'];
 64             $name_column = 'username';
 65             $hook_to_run_before = 'Psa_Hook_Before_User_Delete';
 66             $hook_to_run_after = 'Psa_Hook_After_User_Delete';
 67         }
 68         // work with groups
 69         else if($user_or_group == 2){
 70             $table = $PSA_CFG['database']['table']['group'];
 71             $name_column = 'name';
 72             $hook_to_run_before = 'Psa_Hook_Before_Group_Delete';
 73             $hook_to_run_after = 'Psa_Hook_After_Group_Delete';
 74         }
 75 
 76         // delete users or groups
 77         foreach ($id as $id_key => &$id_value){
 78 
 79             // delete all users or groups
 80             if($id_value == 'all')
 81                 $sql = "DELETE FROM {$table}";
 82             // delete user or groups by id
 83             else if(psa_is_int($id_value)){
 84                 $sql = "DELETE FROM {$table} WHERE id = '$id_value'";
 85             }
 86             // delete user or groups by name
 87             else{
 88                 $sql = "DELETE FROM {$table} WHERE $name_column = " . $psa_database->escape($id_value);
 89             }
 90 
 91             // run Psa_Hook_Before_[user/group]_Delete hooks
 92             psa_run_hooks(array($hook_to_run_before => array('psa_main' => array($id_value))),'by_type','no_unregistered_warning');
 93 
 94             // run query against the database
 95             $psa_database->query($sql);
 96 
 97             // if no rows affected
 98             if($psa_database->affected_rows() <= 0){
 99                 $failed = 1;
100                 $log_data['message']  = 'Unable to delete ' . (($user_or_group == 1) ? 'user' : 'group') . ". Maybe does not exists.";
101             }
102             // run Psa_Hook_After_[user/group]_Delete hooks
103             else{
104                 psa_run_hooks(array($hook_to_run_after => array('psa_main' => array($id_value))),'by_type','no_unregistered_warning');
105                 $log_data['message']  = (($user_or_group == 1) ? 'User' : 'Group') . ' deleted';
106             }
107 
108             // logging
109             if($PSA_CFG['logging']['max_log_level'] >= 2){
110                 // parameters for Psa_Logger::log() method
111                 $log_data['function'] = __FUNCTION__;
112                 $log_data['level']    = 2;
113 
114                 if($user_or_group == 1){
115                     if(psa_is_int($id_value))
116                         $log_data['user_id'] = $id_value;
117                     else
118                         $log_data['username'] = $id_value;
119                 }
120                 else if($user_or_group == 2){
121                     if(psa_is_int($id_value))
122                         $log_data['group_id'] = $id_value;
123                     else
124                         $log_data['groupname'] = $id_value;
125                 }
126 
127                 Psa_Logger::get_instance()->log($log_data);
128             }
129         }
130 
131         if($failed)
132             return -1;
133         else
134             return 1;
135     }
136 
137     return 0;
138 }

193 /**
194  * Executes (runs) hooks.
195  *
196  * This function is used for calling hooks.
197  * It includes hooks files, makes a new instances of hook objects named like hook class
198  * and calls member methods. It can call class methods in two different ways depending on second argument.
199  *
200  * If the second argument is true hooks are called <i>by type</i> (hook type is the name of the class it extends)
201  * and in this case first argument should be an array with elements like in example below:
202  * <code>
203  * // $run is array that will be argument for psa_run_hooks() function
204  * // Structure should be: $run['hook_type']['hook_method'] = array('method_argument1','method_argument2',...);
205  * $run['Psa_Hook_Before_User_Delete']['psa_main'] = array(177);
206  * psa_run_hooks($run, 'by_type');
207  * </code>
208  * This example calls method <kbd>psa_main</kbd> with one argument (177) from all registered
209  * <kbd>Psa_Hook_Before_User_Delete</kbd> hooks.
210  *
211  * If the second argument is false (default) hooks are called <i>by name</i> (hook name is same
212  * as it's class name) and in this case first argument should be array with elements
213  * like in this example:
214  * <code>
215  * // $run is array that will be argument for psa_run_hooks() function
216  * // Structure should be: $run['class_name']['hook_method'] = array('method_argument1','method_argument2',...);
217  * $run['Article']['load'] = array(14);
218  * $run['News']['open'] = array();
219  * psa_run_hooks($run);
220  * </code>
221  * This example calls method <kbd>load</kbd> with one argument (14) from the <kbd>Article</kbd> class
222  * and method <kbd>open</kbd> with no arguments from the <kbd>News</kbd> class.
223  *
224  * This function will throw {@link Psa_Exception} on error.
225  *
226  * @param array $run_data Array with data what to run.
227  * @param bool $by_type If true hooks will be called <i>by type</i> otherwise will be called <i>by name</i>.
228  * @param bool $disable_unregistered_exception Exception that hook is not registered will not be thrown.
229  * This parameter can be set to true when hook doesn't have to be registered like
230  * {@link Psa_Hook_After_User_Create}.
231  * @return int 1:success all hooks executed, -1:success, but not with all hooks because some are
232  * probably not registered. -1 can be returned only when third argument is set to true otherwise
233  * exception will be thrown.
234  * @see Psa_Files::register()
235  * @throws Psa_Exception
236  */
237 function psa_run_hooks($run_data, $by_type = false, $disable_unregistered_exception = false){
238 
239     // Disable hooks globally. I use this while testing.
240     $psa_registry = Psa_Registry::get_instance();
241     if(isset($psa_registry->psa_disable_hooks) && $psa_registry->psa_disable_hooks)
242         return;
243 
244     // files object
245     $psa_files = Psa_Files::get_instance();
246 
247 
248     // die if files are not registered or not retrieved from session or database
249     if(!$psa_files->files_data or !is_array($psa_files->files_data['class_paths'])){
250         $msg = 'No files to run. Files are not registered or data not retrieved from session or database';
251         throw new Psa_Fatal_Error_Exception($msg, $msg, 16);
252     }
253 
254 
255     // exit if $run_data is not array
256     if(!is_array($run_data)){
257         throw new Psa_Exception('Invalid first argument for psa_run_hooks() function. Not an array.', 11);
258     }
259 
260 
261     // call hooks
262     if($by_type){
263 
264         $file_type = key($run_data);
265 
266         // if file is registered
267         if(isset($psa_files->files_data['hooks'][$file_type]))
268             $files_data = &$psa_files->files_data['hooks'][$file_type];
269         else{
270             if(!$disable_unregistered_exception){
271                 throw new Psa_Exception("Trying to run unregistered hook $file_type. Try to register files and delete session (cookies) if you store registered files data in session.", 12);
272             }
273             else
274                 return 0;
275         }
276     }
277     // call hooks by name (hook class name)
278     else{
279         $files_data = &$run_data;
280     }
281 
282 
283     // set default return value
284     $return = 1;
285 
286 
287     $router = new Psa_Router();
288 
289 
290     // $files_data is array with data used to instance classes and run methods (class name, method name, file name)
291     foreach ($files_data as $files_data_key => &$files_data_value){
292 
293         // hooks
294         if($by_type){
295             $include_file = $files_data_value;
296             $file_class = $files_data_key;
297             $file_methods_data = &$run_data[$file_type]; // array of method names and arguments
298         }
299         // if we call files by name
300         else{
301 
302             // if file is not registered
303             if(!isset($psa_files->files_data['class_paths'][$files_data_key])){
304 
305                 if(!$disable_unregistered_exception){
306                     throw new Psa_Exception("Trying to run unregistered class $files_data_key. Try to register files and delete session (cookies) if you store registered file data in session.", 13);
307                 }
308                 else{
309                     $return = -1;
310 
311                     continue;
312                 }
313             }
314 
315             $include_file = $psa_files->files_data['class_paths'][$files_data_key];
316             $file_class = $files_data_key;
317             $file_methods_data = &$files_data_value; // array of method names and arguments
318         }
319 
320         // call object methods
321         if(is_array($file_methods_data)){
322             foreach ($file_methods_data as $file_method_name => $file_method_args){
323                 $router->dispach($file_class, $file_method_name, $file_method_args, $disable_unregistered_exception);
324             }
325         }
326     }
327 
328     return $return;
329 }

332 /**
333  * This function is registered with PHP <kbd>spl_autoload_register()</kbd> function as <kbd>__autoload()</kbd> implementation
334  * used to auto include files.
335  *
336  * Thus if you want to extend some class you don't have to include its file, it will be included automatically.
337  *
338  * <b>Note:</b> Class autoloading will be working only in folders specified with
339  * <var>$PSA_CFG['folders']['autoload'][]</var> and <var>$PSA_CFG['folders']['hook_autoload'][]</var> configuration
340  * options. Also file registration must be invoked before to generate <kbd>autoload_data.php</kbd> file. There is CLI
341  * <kbd>register_files.php</kbd> script for that.
342  *
343  * For example, you can just extends Psa_Mode class without including it's file:
344  * <code>
345  * <?php
346  * class My_Model extends Psa_Model{
347  *  function some_function(){
348  *      // ... something here
349  *  }
350  * }
351  * ?>
352  * </code>
353  *
354  * @see Psa_Files::register()
355  */
356 function psa_autoload($class_name){
357 
358     static $psa_files = null;
359 
360     // read data from generated autoload_data.php file
361     if(!$psa_files){
362         $psa_files = Psa_Files::get_instance();
363         $psa_files->set_data();
364     }
365 
366     // try to include registered class
367     if($psa_files->files_data && array_key_exists($class_name, $psa_files->files_data['class_paths'])){
368         include_once $psa_files->files_data['class_paths'][$class_name];
369         return;
370     }
371 }

374 /**
375  * You can use this function instead of <kbd>echo</kbd> or <kbd>print_r</kbd> functions during development.
376  *
377  * <b>Example:</b>
378  *
379  * <code>
380  * // print $array
381  * prs($array);
382  * </code>
383  *
384  * @param mixed $str
385  * @param bool $return_only
386  */
387 function prs($str, $return_only = false){
388     $return = '<pre>' . htmlspecialchars (print_r($str, TRUE)) . '</pre>';
389 
390     if($return_only)
391         return $return;
392 
393     echo $return;
394     flush ();
395 }

398 /**
399  * Checks if user is a member of the group.
400  *
401  * It simply tries to find a matching row in <i>psa_user_in_group</i> database table.
402  *
403  * <b>Example:</b>
404  *
405  * <code>
406  * if(psa_is_user_in_group(2, 55))
407  *  echo 'User with ID 2 is in the group with ID 55';
408  * </code>
409  *
410  * @param int $user user id
411  * @param int $group group id
412  * @return int 1 if user is a member of the group, otherwise 0
413  */
414 function psa_is_user_in_group($user, $group){
415 
416     $PSA_CFG = Psa_Registry::get_instance()->PSA_CFG;
417 
418     $user = (int)$user;
419     $group = (int)$group;
420 
421     if($user && $group){
422 
423         $sql = "SELECT * FROM {$PSA_CFG['database']['table']['user_in_group']} WHERE user_id = '$user' AND group_id = '$group'";
424 
425         $row = Psa_Registry::get_instance()->psa_database->fetch_row($sql);
426 
427         if(isset($row['user_id']) && $row['user_id'])
428             return 1;
429     }
430 
431     return 0;
432 }

435 /**
436  * Adds given path to PHP <var>include_path</var> configuration directive.
437  *
438  * It can be useful when you work with some externals libraries.
439  *
440  * <b>Example:</b>
441  *
442  * <code>
443  * psa_add_include_path('/usr/share/somefolder');
444  * </code>
445  *
446  * @param string $path path to add to include path
447  * @return string|bool returns the old include_path on success or FALSE on failure.
448  * @see http://www.php.net/manual/en/ini.core.php#ini.include-path
449  */
450 function psa_add_include_path($path){
451 
452     if($path){
453         return set_include_path(get_include_path() . PATH_SEPARATOR . $path);
454     }
455 
456     return false;
457 }
